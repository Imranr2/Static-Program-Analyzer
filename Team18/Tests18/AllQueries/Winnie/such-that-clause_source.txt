procedure Main {
    read x;
    read y;

    if (x < 0) then {
        read y;
        print x;
    } else {
        read x;
        print y;
    }

    read z;
    while (z < 10) {
        print z;
        read x;
        y = z + x;
    }

    if (z < x + y) then {
        if (x < y + 10) then {
            while (x < y + 50 - z) {
                x = x + 1;
                print x;
            }
            read z;
        } else {
            while (z < x * 2) {
                read y;
                z = y - z;
                print y;
            }
            x = z + 5;
        }
    } else {
        print z;
        read z;
    }

    print x;
    y = z / x * 10;
    read z;

    while ((x < y) && (z > x)) {
        read x;
        if (z < 10) then {
            y = z + 1;
            print x;
            z = z * z / y;
        } else {
            print z;
            read y;
            y = z / x * 2;
            print z;
        }
        x = y + z;
        print x;
    }
    read z;
}
